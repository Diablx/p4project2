CREATE TABLE [dbo].[Level] (
    [LevelNo]       INT IDENTITY (1, 1) NOT NULL,
    [Health]        INT NOT NULL,
    [Mana]          INT NOT NULL,
    [ExperienceReq] INT NOT NULL,
    CONSTRAINT [PK_Level] PRIMARY KEY CLUSTERED ([LevelNo] ASC)
);
